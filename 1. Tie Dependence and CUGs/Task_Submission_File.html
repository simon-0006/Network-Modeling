# Keep original node labels and normalize possible "X" prefixes in numeric IDs
row_ids <- rownames(raw_net)
col_ids <- colnames(raw_net)

# Convert to numeric matrix and align to common IDs on both axes
A <- as.matrix(raw_net)
storage.mode(A) <- "numeric"
rownames(A) <- row_ids
colnames(A) <- col_ids

common_ids <- intersect(row_ids, col_ids)
A <- A[common_ids, common_ids, drop = FALSE]

# 2) Identify nodes affected by missing tie data
nodes_drop <- rownames(A)[rowSums(is.na(A)) > 0]

# 3) Remove affected nodes from network (row + column)
A_clean <- A[!(rownames(A) %in% nodes_drop), !(colnames(A) %in% nodes_drop), drop = FALSE]

# 4) Final checks
stopifnot(nrow(A_clean) == ncol(A_clean))
stopifnot(all(A_clean %in% c(0, 1)))

Task 1.1

Compute probability \(p\) of observing a tie if you choose one possible dyad at random

nodes = nrow(raw_net) # We work with the raw input (not the cleaned A_clean)
ties  = sum(raw_net, na.rm = TRUE)

# P[tie] = |ties| / |total number of possible ties|
n_dyads = (nodes) * (nodes - 1) / 2
p = ties / n_dyads
cat("Probability[Observing a Tie] = ", p, "\n")

Task 1.2

Compute the probabilities \(p_M, p_A, p_N\) of observing a mutual, asymmetric and null dyad

# Convert to numeric matrix, treating NA as missing
A_raw <- as.matrix(raw_net)
storage.mode(A_raw) <- "numeric"

# Get upper triangle indices (each dyad counted once, i < j)
upper <- upper.tri(A_raw)
ij <- A_raw[upper]          # tie from i -> j
ji <- t(A_raw)[upper]       # tie from j -> i

n_mutual     <- sum(ij == 1 & ji == 1, na.rm = TRUE)
n_asymmetric <- sum(xor(ij == 1, ji == 1), na.rm = TRUE)
n_null       <- sum(ij == 0 & ji == 0, na.rm = TRUE)
n_nan        <- n_dyads - (n_mutual + n_asymmetric + n_null)

p_M <- n_mutual     / n_dyads
p_A <- n_asymmetric / n_dyads
p_N <- n_null       / n_dyads
p_Na <- n_nan / n_dyads

knitr::kable(
  data.frame(
    Dyad        = c("Mutual", "Asymmetric", "Null", "Rest (NA)", "Total"),
    Count       = c(n_mutual, n_asymmetric, n_null, n_nan, n_dyads),
    Probability = c(p_M, p_A, p_N, p_Na, p_M + p_A + p_N + p_Na)
  ),
  digits = 4,

)

Task 1.3

What would be the value of the probabilities in (2) if we assume tie independence and that the probability of observing a tie takes the value \(p\) computed in (1)?

# T : Tie
# N : No Tie
# 
# p[T] = p
# p[N] = 1 - p
# 
#      . 
#     / \ 
#   T     N
#  /\     /\   
# t  N   T  N

p_uni_M <- p * p
p_uni_A <- 2 * p * (1 - p)
p_uni_N <- (1 - p) * (1 - p)
