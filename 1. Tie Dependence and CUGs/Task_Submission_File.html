# Keep original node labels and normalize possible "X" prefixes in numeric IDs
row_ids <- rownames(raw_net)
col_ids <- colnames(raw_net)

# Convert to numeric matrix and align to common IDs on both axes
A_raw <- as.matrix(raw_net)
storage.mode(A_raw) <- "numeric"
rownames(A_raw) <- row_ids
colnames(A_raw) <- col_ids

common_ids <- intersect(row_ids, col_ids)
A_raw <- A_raw[common_ids, common_ids, drop = FALSE]

# 2) Identify nodes affected by missing tie data
nodes_drop <- rownames(A_raw)[rowSums(is.na(A_raw)) > 0]

# 3) Remove affected nodes from network (row + column)
A_clean <- A_raw[!(rownames(A_raw) %in% nodes_drop), !(colnames(A_raw) %in% nodes_drop), drop = FALSE]

# 4) Final checks
stopifnot(nrow(A_clean) == ncol(A_clean))
stopifnot(all(A_clean %in% c(0, 1)))

Compute probability \(p\) of observing a tie if you choose one possible dyad at random

nodes = n
ties  = m

# P[tie] = |ties| / |total number of possible ties|
n_dyads = nodes * (nodes - 1) / 2
n_ties = nodes * (nodes - 1)
p = ties / n_ties

Task 1.3

What would be the value of the probabilities in (2) if we assume tie independence and that the probability of observing a tie takes the value \(p\) computed in (1)?

# T : Tie
# N : No Tie
# 
# p[T] = p
# p[N] = 1 - p
# 
#      . 
#     / \ 
#   T     N
#  /\     /\   
# t  N   T  N

p_uni_M <- p * p
p_uni_A <- 2 * p * (1 - p)
p_uni_N <- (1 - p) * (1 - p)

Task 1.4

Given the values obtained in (2) and (3), would it be reasonable to assume tie independence when performing further analysis on the observed network? Justify

No, it would not be reasonable to assume tie independence.

observed    <- c(p_M, p_A, p_N)
independent <- c(p_uni_M, p_uni_A, p_uni_N)

knitr::kable(
  data.frame(
    Dyad        = c("Mutual", "Asymmetric", "Null"),
    Observed    = observed,
    Independent = independent,
    Factor      = observed / independent

  digits = 4,
  caption = "Observed vs. Independence Assumption"

Observed vs. Independence Assumption
Dyad Observed Independent Factor
Mutual 0.1818 0.0841 2.1613

As shown in the table we observe roughly twice as many mutual ties as we would by assuming independence. On the other hand asymmetric ties would be overrepresented by the assumption of independence.

Task 1.5

Consider a different model that assumes all nodes have an outdegree of nine (d=9). Compute the probability of a mutual tie assuming tie independence. Compare this probability with the probabilities computed in (2) and (3). Would it be reasonable to assume this model?

m_nine <- n * 9
p_nine <- m_nine / n_ties
cat("Probability[Observing a Tie given (d=9)] = ", p_nine, "\n")
Probability[Observing a Tie given (d=9)] =  0.4285714 
p_nine_M <- p_nine * p_nine
p_nine_A <- 2 * p_nine * (1 - p_nine)
p_nine_N <- (1 - p_nine) * (1 - p_nine)

knitr::kable(
  data.frame(
    Dyad        = c("Mutual", "Asymmetric", "Null"),
    Observed    = c(p_M, p_A, p_N),
    `d=9`       = c(p_nine_M, p_nine_A, p_nine_N),
    Factor      = c(p_M, p_A, p_N) / c(p_nine_M, p_nine_A, p_nine_N)
  ),
  check.names = FALSE,
  digits = 4,
  caption = "Observed vs. d=9 Independence Model"
)
Observed vs. d=9 Independence Model
Dyad Observed d.9 Factor
Mutual 0.1818 0.1837 0.9899
Asymmetric 0.2165 0.4898 0.4419
Null 0.6017 0.3265 1.8428

Comment about this model:
As we can see this model aligns pretty well the probability of observing a mutual dyad. On the other hand are asymmetric and null dyads not closely aligned.

If we think about how many ties there are: \(22 \cdot 9 = 198\), we see it is a lot more than the actual number of ties which is \(m = 134\).
Logically (by assuming tie independence) we would need to pick \(d = m/n = 134/22 \approx 6\).

Conclusion: We can match one statistic (in that case mutuality) by tuning a parameter, but the overall dyad distribution still doesn’t fit. This means this model isn’t adequate either.
